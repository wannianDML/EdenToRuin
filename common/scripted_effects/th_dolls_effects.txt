effect_月度dolls产能结算 = {
	if = {
		limit = {
			not = { has_variable = var_积累的dolls产能 }
		}
		set_variable = {
			name = var_积累的dolls产能
			value = 0
		}
	}

	change_variable = {
		name = var_积累的dolls产能
		add = var_月度dolls产能
	}

	change_variable = {
		name = var_积累的dolls产能
		min = var_dolls产能存储上限
	}
}
